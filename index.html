async function loadQuestions() { const container = document.getElementById('questionContainer'); container.innerHTML = ""; try { const provider = new ethers.providers.Web3Provider(window.ethereum); const contract = new ethers.Contract(contractAddress, abi, provider); const count = await contract.getMarketCount(); for(let i = 0; i < count.toNumber(); i++) { const data = await contract.markets(i); // কার্ডের ভেতরে বাটন যোগ করা হয়েছে container.innerHTML += `

${data.question}

Yes: ${data.totalYes} | No: ${data.totalNo}
`; } } catch(e) { console.error("Error loading:", e); } } // নতুন বেট ফাংশন async function bet(marketId, prediction) { const provider = new ethers.providers.Web3Provider(window.ethereum); const signer = provider.getSigner(); const contract = new ethers.Contract(contractAddress, abi, signer); try { const tx = await contract.bet(marketId, prediction, { value: ethers.utils.parseEther("0.001") }); await tx.wait(); alert("Bet Placed!"); loadQuestions(); } catch(e) { alert("Error: " + e.message); } }
